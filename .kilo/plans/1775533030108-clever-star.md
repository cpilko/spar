        Tool(

            description="Fetch a URL and extract content",
            instructions="Fetch the URL and return the page content.",

async def ask_agent(system_prompt: str, prompt: str, label: str, color: str) -> str:
    """Fire a query using OpenAI Agents SDK, stream tool usage, collect final text."""
    agent_type = {
        "RAZOR": "razor", "EMBER": "ember", "JUDGE": "judge",
        "VIPER": "viper", "PITCH": "pitch", "SCOUT": "scout", "ASK": "ask"
    }.get(label.upper(), "ember")

    agent_map = {
        "razor": create_razor_agent,
        "ember": create_ember_agent,
        "judge": create_judge_agent,
        "viper": create_viper_agent,
        "pitch": create_pitch_agent,
        "scout": create_scout_agent,
        "ask": create_ask_agent,
    }

    agent_fn = agent_map.get(agent_type, create_ember_agent)
    agent = agent_fn()

    result = await Runner.run(agent, prompt)

    return result.final_output if hasattr(result, 'final_output') else str(result)

## Key Differences Summary

| Aspect | Claude SDK | OpenAI SDK |
|--------|-----------|------------|
| Agent Definition | `ClaudeAgentOptions` with `system_prompt` | `Agent` with `instructions` |
| Execution | `async for message in query()` | `Runner.run(agent, input)` |
| Tool Handling | Auto-processed via `ToolUseBlock` | Explicit `Tool` definitions + function implementations |
| Streaming | `AssistantMessage` with `TextBlock`/`ToolUseBlock` | `AsyncResult` with streaming |
| Reasoning | `ThinkingConfigEnabled` | Native for `o1`/`o3` models |
| Permissions | `permission_mode` | Not applicable (uses API key) |

## Risks & Considerations

1. **Tool Implementations**: The Claude SDK likely had built-in tool implementations. You must provide equivalent implementations for OpenAI (either Python functions, MCP servers, or hosted tools).
2. **Streaming UX**: The current code streams tool usage live. OpenAI SDK's streaming requires different handling.
3. **Model Differences**: Claude Opus 4 and GPT-4o/o1 have different capabilities and response formats. Prompt tuning may be needed.
4. **Cost**: OpenAI API pricing differs from Anthropic. Monitor usage.
